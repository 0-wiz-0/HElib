29 class CubeSignature {
30 private:
31  Vec<long> dims; // dims[i] is the size along the i'th diemnsion
32  Vec<long> prods; // prods[i] = \prod_{j=i}^{n-1} dims[i]
33  long ndims;
34  long size;
35 
36 public:
37  CubeSignature(): ndims(0) {} // a NULL signature
38 
39  CubeSignature(const Vec<long>& _dims): ndims(0) { initSignature(_dims); }
40 
41  /* When we get C++11 support, we could #include <initializer_list>
42  * and then do e.g., CubeSignature s {1,2,3};
43  CubeSignature(std::initializer_list<long> _dims)
44  {
45  dims.SetLength(_dims.size());
46  long i;
47  std::initializer_list<long>::iterator it;
48  for (i=0, it=_dims.begin(); it!=_dims.end(); ++i, ++it)
49  dims[i] = *it;
50 
51  [...] // continue as above, ndims = dims.length() etc.
52  }
53  **********************************************************/

55  void initSignature(const Vec<long>& _dims)
56  {
57  assert(ndims == 0); // can only initialize a NULL signature
58  dims = _dims;
59  ndims = dims.length();
60  assert(ndims > 0);
61 
62  prods.SetLength(ndims+1);
63  prods[ndims] = 1;
64  for (long i = ndims-1; i >= 0; i--) {
65  assert(dims[i] > 0);
66  prods[i] = dims[i]*prods[i+1];
67  }
68  size = prods[0];
69  }
70 
72  long getSize() const { return size; }
73 
75  long getNumDims() const { return ndims; }
76 
78  long getDim(long d) const { return dims.at(d); }
79 
81  long getProd(long d) const { return prods.at(d);}
82 
84  long getProd(long from, long to) const
85  { return prods.at(from)/prods.at(to); }
86 
88  long getCoord(long i, long d) const {
89  assert(i >= 0 && i < size);
90 
91  return (i % prods.at(d)) / prods.at(d+1);
92  }
93 
95  long addCoord(long i, long d, long offset) const {
96  assert(i >= 0 && i < size);
97 
98  offset = offset % dims.at(d);
99  if (offset < 0) offset += dims.at(d);
100 
101  long i_d = getCoord(i, d);
102  long i_d1 = (i_d + offset) % dims.at(d);
103 
104  long i1 = i + (i_d1 - i_d) * prods.at(d+1);
105 
106  return i1;
107  }
108 
110  long numSlices(long d=1) const { return getProd(0, d); }
111 
113  long sliceSize(long d=1) const { return getProd(d); }
114 
116  long numCols() const { return getProd(1); }
117
