188  void parse(int argc, char **argv)
189  {
190  for (long i = 1; i < argc; i++) {
191  const char *x = argv[i];
192  long j = 0;
193  while (x[j] != '=' && x[j] != '\0') j++;
194  if (x[j] == '\0') usage(argv[0]);
195  string name(x, j);
196  const char *s = x+j+1;
197 
198  shared_ptr<ArgProcessor> ap = map[name];
199  if (!ap) return usage(argv[0]);
200  if (!ap->process(s)) usage(argv[0]);
201  }
202  }

482 template <class T, bool maxFlag>
483 long argminmax(vector<T>& v)
484 {
485  if (v.size()<1) return -1; // error: this is an empty array
486  unsigned long idx = 0;
487  T target = v[0];
488  for (unsigned long i=1; i<v.size(); i++)
489  if (maxFlag) { if (v[i] > target) { target = v[i]; idx = i;} }
490  else { if (v[i] < target) { target = v[i]; idx = i;} }
491  return (long) idx;
492 }

587 template<class T> void reverse(Vec<T>& v, long lo, long hi)
588 {
589  long n = v.length();
590  assert(lo >= 0 && lo <= hi && hi < n);
591 
592  if (lo >= hi) return;
593 
594  for (long i = lo, j = hi; i < j; i++, j--) swap(v[i], v[j]);
595 }

598 // Example: rotate by 1 means [0 1 2 3] -> [3 0 1 2]
599 // rotate by -1 means [0 1 2 3] -> [1 2 3 0]
600 template<class T> void rotate(Vec<T>& v, long k)
601 {
602  long n = v.length();
603  if (n <= 1) return;
604 
605  k %= n;
606  if (k < 0) k += n;
607 
608  if (k == 0) return;
609 
610  reverse(v, 0, n-1);
611  reverse(v, 0, k-1);
612  reverse(v, k, n-1);
613 }
