143  Vec< Vec<short> > level;
144  // there are 2*k - 1 levels, each wity n nodes.
145  // level[i][j] is 0, 1, or -1,
146  // which designates an edge from node j at level i
147  // to node j + level[i][j]*shamt(i) at level i+1
148 
149  GeneralBenesNetwork(); // default constructor disabled
150 
151 public:
154  static long depth(long n) {
155  long k = 1;
156  while ((1L << k) < n) k++;
157  return k;
158  }

162  static long levelToDepthMap(long n, long k, long i) {
163  assert(i >= 0 && i < 2*k-1);
164  return (k-1) - labs((k-1)-i);
165  }

169  static long shamt(long n, long k, long i) {
170  long d = levelToDepthMap(n, k, i);

313 template <class T>
314 long FullBinaryTree<T>::addChildren(long prntIdx,
315  const T& leftData, const T& rightData)
316 {
317  assert (prntIdx >= 0 && prntIdx < (long)(nodes.size()));
318 
319  // If parent is a leaf, add to it two children
320  if (nodes[prntIdx].leftChild==-1 && nodes[prntIdx].rightChild==-1) {
321  long childIdx = nodes.size();
322  TreeNode<T> n1(leftData);
323  nodes.push_back(n1); // add left child to vector
324  TreeNode<T> n2(rightData);
325  nodes.push_back(n2);// add right child to vector
326 
327  TreeNode<T>& parent = nodes[prntIdx];
328  TreeNode<T>& left = nodes[childIdx];
329  TreeNode<T>& right = nodes[childIdx+1];
330 
331  parent.leftChild = childIdx; // point to children from parent
332  parent.rightChild= childIdx+1;
333  left.parent = right.parent = prntIdx; // point to parent from children
334 
335  // remove parent and insert children to the linked list of leaves
336  left.prev = parent.prev;
337  left.next = childIdx+1;
338  right.prev = childIdx;
339  right.next = parent.next;
340  if (parent.prev>=0) { // parent was not the 1st leaf
341  nodes[parent.prev].next = childIdx;
342  parent.prev = -1;
343  }
344  else // parent was the first leaf, now its left child is 1st
345  frstLeaf = childIdx;
346 
347  if (parent.next>=0) { // parent was not the last leaf
348  nodes[parent.next].prev = childIdx+1;
349  parent.next = -1;
350  }
351  else // parent was the last leaf, now its left child is last
352  lstLeaf = childIdx+1;
353 
354  nLeaves++; // we replaced a leaf by a parent w/ two leaves
355  }
356  else { // parent is not a leaf, update the two children
357  TreeNode<T>& parent = nodes[prntIdx];
358  assert(parent.leftChild>=0 && parent.rightChild>=0);
359 
360  TreeNode<T>& left = nodes[parent.leftChild];
361  TreeNode<T>& right = nodes[parent.rightChild];
362  left.data = leftData;
363  right.data = rightData;
364  }
365  return nodes[prntIdx].leftChild;
366 }
