144  bool mul(const SKHandle& a, const SKHandle& b)
145  {
146  // If either of the inputs is one, the output equals to the other input
147  if (a.isOne()) {
148  *this = b;
149  return (b.secretKeyID >= 0);
150  }
151  if (b.isOne()) {
152  *this = a;
153  return (a.secretKeyID >= 0);
154  }
155 
156  if (a.secretKeyID == -1 || b.secretKeyID == -1) {
157  secretKeyID = -1; // -1 will be used to indicate an "error state"
158  return false;
159  }
160 
161  if (a.secretKeyID != b.secretKeyID) {
162  secretKeyID = -1;
163  return false;
164  }
165 
166  if (a.powerOfX != b.powerOfX) {
167  secretKeyID = -1;
168  return false;
169  }
170 
171  secretKeyID = a.secretKeyID;
172  powerOfX = a.powerOfX;
173  powerOfS = a.powerOfS + b.powerOfS;
174  return true;
175  }
