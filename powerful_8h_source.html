96  void initPConv(const PowerfulTranslationIndexes& ind)
97  {
98  if (indexes!=NULL) return; // cannot re-initialize a non-NULL object
99  indexes = &ind;
100 
101  cycVec_p.SetLength(ind.cycVec.length());
102  zzpContext.save(); // store the current modulus
103  for (long i=0; i<ind.cycVec.length(); i++) {
104  cycVec_p[i] = conv<zz_pX>(ind.cycVec[i]); // convert to zz_pXModulus
105  }
106  phimX_p = conv<zz_pX>(ind.phimX); // convert to zz_pXModulus
107  }
108 
109  void restoreModulus() const { zzpContext.restore(); }
